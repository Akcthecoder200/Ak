#include <Wire.h>
#include <Adafruit_BMP085.h>
#include <LiquidCrystal_I2C.h>

// Create instances
Adafruit_BMP085 bmp;
LiquidCrystal_I2C lcd(0x27, 16, 2); // Use 0x3F if 0x27 doesn't work

void setup() {
  lcd.init();
  lcd.backlight();

  lcd.setCursor(0, 0);
  lcd.print("BMP180 Test");

  if (!bmp.begin()) {
    lcd.setCursor(0, 1);
    lcd.print("Sensor Error!");
    while (1); // Stop execution if sensor not found
  }

  delay(2000);
  lcd.clear();
}

void loop() {
  float temp = bmp.readTemperature();          // in Â°C
  float pressure = bmp.readPressure() / 100.0; // in hPa

  lcd.setCursor(0, 0);
  lcd.print("Temp: ");
  lcd.print(temp);
  lcd.print(" C  ");

  lcd.setCursor(0, 1);
  lcd.print("Pres: ");
  lcd.print(pressure);
  lcd.print(" hPa");

  delay(2000); // Update every 2 seconds
}
